Programa para calcular crescimento de uma população

Programa para calcular crescimento de uma população

Programa para calcular o crescimento da população de uma determinada cidade, usando a estrtutura de repetição while
def calcula_populacao_anos(populacao_a, taxa_a, populacao_b, taxa_b):
anos = 0
while populacao_a <= populacao_b:
populacao_a *= (1 + taxa_a / 100)
populacao_b *= (1 + taxa_b / 100)
anos += 1
return anos

populacao_a = 80000
taxa_a = 3

populacao_b = 200000
taxa_b = 1.5

numero_anos = calcula_populacao_anos(populacao_a, taxa_a, populacao_b, taxa_b)

print(f"Serão necessários {numero_anos} anos para que a população do país A ultrapasse ou iguale a população do país B.")
